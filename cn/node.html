For Node.js

nodetpl 除了用于一般的 web 项目中，还可以用在 node.js。

安装 nodetpl

npm install nodetpl -g

基本语法

var fs = require('fs');
var nodetpl = require('nodetpl');

var filepath = '/usr/local/nodetpl/test.tpl';
var data = {
  name: 'zhangsan',
  age: 18
};

fs.readFile(filepath, 'utf-8', function(err, content) {
  if (err) {
    throw err;
  } else {
    nodetpl.render(content, data, function(d){
      console.log(d);
    });
  }
});

For Express.js

app.set('view engine', 'tpl');
app.set('views', __dirname + '/views');
app.engine('tpl', require('nodetpl').express.render);

app.get('/', function(req, res){
  res.render('index', {
    "user": "Tom",
    "favor": [ "Apple", "Orange", "Bananer" ]
  }, function(err, html){
    res.writeHead(200, { "Content-Type": "text/html" });
    res.end(html);
  });
});
