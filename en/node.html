For Node.js

nodetpl addition to general web project can also be used in node.js.

Installation nodetpl

npm install nodetpl -g

The basic syntax

var fs = require('fs');
var nodetpl = require('nodetpl');

var filepath = '/usr/local/nodetpl/test.tpl';
var data = {
  name: 'zhangsan',
  age: 18
};

fs.readFile(filepath, 'utf-8', function(err, content) {
  if (err) {
    throw err;
  } else {
    content = nodetpl.render(content, data);
    console.log(content);
  }
});

For Express.js

app.set('view engine', 'tpl');
app.set('views', __dirname + '/views');
app.engine('tpl', require('nodetpl').express.render);

app.get('/', function(req, res){
  res.render('index', {
    "user": "Tom",
    "favor": [ "Apple", "Orange", "Bananer" ]
  }, function(err, html){
    res.writeHead(200, { "Content-Type": "text/html" });
    res.end(html);
  });
});
